def avatar_thumbnail(user = current_user)
  if user.profile.respond_to?(:avatar) && user.profile.avatar.nil? == false
    user.profile.avatar.gsub("-/resize/600x600/", "-/resize/70x70/-/quality/lightest/")
  else
    "default-avatar.jpg"
  end
end
