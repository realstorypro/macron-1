def cast
  subject = if slugged?(entry_class)
    entry_class.friendly.find(params[:subject_id])
  else
    entry_class.find(params[:subject_id])
  end

  render json: { points: current_user.cast_spell!(params[:spell].to_sym, subject) }
end
