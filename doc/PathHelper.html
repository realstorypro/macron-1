def index_path(namespace = nil, component = params[:component])
  path = path_getter(component)
  if path.pluralize(nil) == path
    "#{in_namespace(namespace)}#{path}_path"
  else
    "#{in_namespace(namespace)}#{path}_index_path"
  end
end
